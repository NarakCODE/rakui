## Usage

```tsx title="visually-hidden-input-snippet.tsx"
import { VisuallyHiddenInput } from "@rakui/visually-hidden-input";

export default function Example() {
  return (
    <form>
      <VisuallyHiddenInput name="hidden-field" value="secret-value" />
      <button type="submit">Submit</button>
    </form>
  );
}
```

## Props

### VisuallyHiddenInput

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `name` | `string` | - | Input name attribute |
| `value` | `string` | - | Input value |
| `onChange` | `(e: ChangeEvent<HTMLInputElement>) => void` | - | Change handler |
| `...rest` | `InputHTMLAttributes<HTMLInputElement>` | - | Additional input attributes |

## Examples

### File Upload

```tsx title="visually-hidden-input-snippet.tsx"
import { VisuallyHiddenInput } from "@rakui/visually-hidden-input";
import { useRef } from "react";

export default function FileUpload() {
  const inputRef = useRef<HTMLInputElement>(null);

  return (
    <div>
      <button 
        type="button" 
        onClick={() => inputRef.current?.click()}
      >
        Choose File
      </button>
      <VisuallyHiddenInput
        ref={inputRef}
        type="file"
        name="file"
        onChange={(e) => console.log(e.target.files)}
      />
    </div>
  );
}
```

### Form Token

```tsx title="visually-hidden-input-snippet.tsx"
import { VisuallyHiddenInput } from "@rakui/visually-hidden-input";

export default function SecureForm() {
  return (
    <form action="/submit" method="POST">
      <VisuallyHiddenInput 
        name="csrf_token" 
        value="secure-token-value" 
      />
      <button type="submit">Submit Form</button>
    </form>
  );
}
```
