## Usage

```tsx title="phone-input-demo.tsx"
import { PhoneInput } from "@rakui/phone-input";

export default function Example() {
  return (
    <PhoneInput
      defaultCountry="US"
      onChange={(phone) => console.log(phone)}
    />
  );
}
```
