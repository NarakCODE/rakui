## Usage

```tsx title="tour-demo.tsx"
import { Tour } from "@rakui/tour";

export default function Example() {
  const steps = [
    { target: "#step1", title: "Welcome", content: "This is the first step" },
    { target: "#step2", title: "Features", content: "Explore our features" },
  ];

  return (
    <Tour
      steps={steps}
      isOpen={true}
      onClose={() => {}}
    />
  );
}
```
