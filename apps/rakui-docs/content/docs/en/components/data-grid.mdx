## Usage

### Basic Data Grid

```tsx title="data-grid-example.tsx"
import { DataGrid } from "@/components/ui/data-grid";
import { DataGridKeyboardShortcuts } from "@/components/data-grid/data-grid-keyboard-shortcuts";
import { useDataGrid } from "@/hooks/use-data-grid";

export default function MyDataGrid() {
  const [data, setData] = React.useState(initialData);

  const columns = React.useMemo(() => [
    {
      id: "name",
      accessorKey: "name",
      header: "Name",
      meta: {
        cell: {
          variant: "short-text",
        },
      },
    },
    // ... other columns
  ], []);

  const {
    table,
    editingState,
    setEditingCell,
    updateCell,
    handleKeyDown,
  } = useDataGrid({
    data,
    columns,
    onDataChange: setData,
    getRowId: (row) => row.id,
  });

  return (
    <>
      <DataGridKeyboardShortcuts />
      <DataGrid
        table={table}
        editingState={editingState}
        onCellEdit={updateCell}
        onCellClick={(rowIndex, columnId) => setEditingCell(rowIndex, columnId)}
        onKeyDown={handleKeyDown}
      />
    </>
  );
}
```

### With Row Management

Add and delete rows with callbacks:

```tsx title="data-grid-example.tsx"
const onRowAdd = React.useCallback(() => {
  setData((prev) => [...prev, { id: generateId() }]);

  return {
    rowIndex: data.length,
    columnId: "name", // Focus this column after creating the new row
  };
}, [data.length]);

const {
  table,
  editingState,
  setEditingCell,
  updateCell,
  handleKeyDown,
  deleteRow,
} = useDataGrid({
  data,
  columns,
  onDataChange: setData,
  onRowAdd,
  getRowId: (row) => row.id,
});
```

### With Search

Use `enableSearch` in the keyboard shortcuts component to enable `Cmd/Ctrl+F` search:

```tsx title="data-grid-snippet.tsx"
const {
  table,
  editingState,
  setEditingCell,
  updateCell,
  handleKeyDown,
  searchState,
} = useDataGrid({
  data,
  columns,
  onDataChange: setData,
});

// Pass search state to keyboard shortcuts for proper shortcuts display
<DataGridKeyboardShortcuts enableSearch={!!searchState} />
```

### With Paste Support

Use the `enablePaste` prop to enable pasting from clipboard:

```tsx title="data-grid-snippet.tsx"
const { table, editingState, ...rest } = useDataGrid({
  data,
  columns,
  onDataChange: setData,
  enablePaste: true, // Enable paste (Ctrl/Cmd+V)
});
```

### Read-Only Mode

Use the `readOnly` prop to make the grid read-only (not yet supported — omit `onCellEdit` and `onCellClick` to achieve this):

```tsx title="data-grid-snippet.tsx"
<DataGrid
  table={table}
  editingState={editingState}
  height={400}
/>
```

### RTL Support

Wrap the grid in a container with `dir="rtl"`:

```tsx title="data-grid-snippet.tsx"
import { DataGrid } from "@/components/ui/data-grid";
import { useDataGrid } from "@/hooks/use-data-grid";

export function DataGridRtlDemo() {
  const { table, editingState, setEditingCell, updateCell, handleKeyDown } =
    useDataGrid({
      columns,
      data,
      onDataChange: setData,
    });

  return (
    <div dir="rtl">
      <DataGrid
        table={table}
        editingState={editingState}
        onCellEdit={updateCell}
        onCellClick={(rowIndex, columnId) => setEditingCell(rowIndex, columnId)}
        onKeyDown={handleKeyDown}
      />
    </div>
  );
}
```

### Custom Height

Control the grid height in pixels (default: `400`):

```tsx title="data-grid-snippet.tsx"
<DataGrid
  table={table}
  editingState={editingState}
  onCellEdit={updateCell}
  onCellClick={(rowIndex, columnId) => setEditingCell(rowIndex, columnId)}
  onKeyDown={handleKeyDown}
  height={600}
/>
```

## Cell Variants

The Data Grid supports various cell variants for different data formats:

### Short Text Cell

Single-line text input with inline editing:

```tsx title="data-grid-snippet.tsx"
{
  id: "name",
  accessorKey: "name",
  header: "Name",
  meta: {
    cell: {
      variant: "short-text",
    },
  },
}
```

### Long Text Cell

Multi-line text displayed in a textarea with auto-save:

```tsx title="data-grid-snippet.tsx"
{
  id: "notes",
  accessorKey: "notes",
  header: "Notes",
  meta: {
    cell: {
      variant: "long-text",
    },
  },
}
```

### Number Cell

Numeric input with optional constraints:

```tsx title="data-grid-snippet.tsx"
{
  id: "price",
  accessorKey: "price",
  header: "Price",
  meta: {
    cell: {
      variant: "number",
      min: 0,
      max: 1000,
    },
  },
}
```

### Checkbox Cell

Boolean checkbox for true/false values:

```tsx title="data-grid-snippet.tsx"
{
  id: "isActive",
  accessorKey: "isActive",
  header: "Active",
  meta: {
    cell: {
      variant: "checkbox",
    },
  },
}
```

### Select Cell

Single-select input with predefined options:

```tsx title="data-grid-snippet.tsx"
{
  id: "category",
  accessorKey: "category",
  header: "Category",
  meta: {
    cell: {
      variant: "select",
      options: [
        { label: "Electronics", value: "electronics" },
        { label: "Clothing", value: "clothing" },
        { label: "Books", value: "books" },
      ],
    },
  },
}
```

### Date Cell

Date picker input:

```tsx title="data-grid-snippet.tsx"
{
  id: "startDate",
  accessorKey: "startDate",
  header: "Start Date",
  meta: {
    cell: {
      variant: "date",
    },
  },
}
```

## Row Selection

Row selection is built into `useDataGrid`. The hook automatically includes a `select` column with checkboxes:

```tsx title="data-grid-snippet.tsx"
const {
  table,
  toggleRowSelection,
  selectAllRows,
  clearSelection,
  getSelectedRowIndices,
} = useDataGrid({
  data,
  columns,
  onDataChange: setData,
  getRowId: (row) => row.id,
});

// Get selected row indices
const selectedIndices = getSelectedRowIndices();
```

Pin the built-in select column to the left:

```tsx title="data-grid-snippet.tsx"
useDataGrid({
  data,
  columns,
  initialState: {
    columnPinning: {
      left: ["select"],
    },
  },
});
```

## Column Configuration

### Enabling Filtering and Sorting

Enable filtering and sorting per-column:

```tsx title="data-grid-snippet.tsx"
const columns = React.useMemo(() => [
  {
    id: "name",
    accessorKey: "name",
    header: "Name",
    enableSorting: true,
    enableColumnFilter: true,
    meta: {
      cell: {
        variant: "short-text",
      },
    },
  },
  // ... other columns
], []);
```

Enable at the grid level:

```tsx title="data-grid-snippet.tsx"
useDataGrid({
  data,
  columns,
  enableSorting: true,
  enableFiltering: true,
});
```

### Column Pinning

Pin columns to the left or right:

```tsx title="data-grid-snippet.tsx"
const { table, ...rest } = useDataGrid({
  data,
  columns,
  initialState: {
    columnPinning: {
      left: ["select", "name"], // Pin to left side
      right: ["actions"],       // Pin to right side
    },
  },
});
```

### Initial Sorting

Set the initial sort state:

```tsx title="data-grid-snippet.tsx"
useDataGrid({
  data,
  columns,
  initialState: {
    sorting: [
      { id: "name", desc: false },
    ],
  },
});
```

### Column Minimum Width

Set minimum column width using `minSize`:

```tsx title="data-grid-snippet.tsx"
{
  id: "name",
  accessorKey: "name",
  header: "Name",
  minSize: 180, // Minimum width in pixels
  meta: {
    cell: {
      variant: "short-text",
    },
  },
}
```

### Cell Variants

Configure cell editing behavior through column `meta.cell`:

| Variant | Options | Description |
|---------|---------|-------------|
| `short-text` | — | Single-line text input |
| `long-text` | — | Multi-line textarea |
| `number` | `min`, `max` | Number input with constraints |
| `select` | `options: {label, value}[]` | Dropdown selection |
| `checkbox` | — | Boolean checkbox |
| `date` | — | Date picker input |
