## Examples

### Variants

Use the `variant` prop to change the button style.

```tsx title="button-snippet.tsx"
import { Button } from "@rakui/button";

export function Variants() {
  return (
    <div className="flex flex-wrap gap-3">
      <Button>Primary</Button>
      <Button variant="secondary">Secondary</Button>
      <Button variant="tertiary">Tertiary</Button>
      <Button variant="outline">Outline</Button>
      <Button variant="ghost">Ghost</Button>
      <Button variant="danger">Danger</Button>
      <Button variant="danger-soft">Danger Soft</Button>
    </div>
  );
}
```

### Sizes

Use the `size` prop to change the button size.

```tsx title="button-snippet.tsx"
import { Button } from "@rakui/button";

export function Sizes() {
  return (
    <div className="flex items-center gap-3">
      <Button size="sm">Small</Button>
      <Button size="md">Medium</Button>
      <Button size="lg">Large</Button>
    </div>
  );
}
```


### Icon Only

Use `isIconOnly` prop for buttons that contain only an icon.

```tsx title="button-snippet.tsx"

import { Ellipsis, Settings, Trash2 } from "lucide-react";

export function IconOnly() {
  return (
    <div className="flex gap-3">
      <Button isIconOnly variant="tertiary">
        <Ellipsis />
      </Button>
      <Button isIconOnly variant="secondary">
        <Settings />
      </Button>
      <Button isIconOnly variant="danger">
        <Trash2 />
      </Button>
    </div>
  );
}
```

### Loading State

Use `isPending` prop to show a loading state. Combine with render props for dynamic content.

```tsx title="button-example.tsx"
"use client";

import { Button } from "@rakui/button";
import { LoaderCircle, Paperclip } from "lucide-react";
import { useState } from "react";

export function LoadingState() {
  const [isLoading, setIsLoading] = useState(false);

  const handlePress = () => {
    setIsLoading(true);
    setTimeout(() => setIsLoading(false), 2000);
  };

  return (
    <Button isPending={isLoading} onPress={handlePress}>
      {({ isPending }) => (
        <>
          {isPending ? <LoaderCircle className="animate-spin" /> : <Paperclip />}
          {isPending ? "Uploading..." : "Upload File"}
        </>
      )}
    </Button>
  );
}
```

### Full Width

Use `fullWidth` prop to make the button take the full width of its container.

```tsx title="button-snippet.tsx"
import { Button } from "@rakui/button";
import { Plus } from "lucide-react";

export function FullWidth() {
  return (
    <div className="w-full max-w-xs space-y-3">
      <Button fullWidth>Primary Button</Button>
      <Button fullWidth>
        <Plus />
        With Icon
      </Button>
    </div>
  );
}
```

### Custom Render Function

Use the `render` prop to customize the rendered element.

```tsx title="button-snippet.tsx"
"use client";

import { Button } from "@rakui/button";

export function CustomRenderFunction() {
  return (
    <Button
      render={(props, { isPressed }) => (
        <button {...props} data-custom={isPressed ? "pressed" : "idle"} />
      )}
    >
      Press me
    </Button>
  );
}
```

### Adding Custom Variants

Extend the Button component with your own custom variants:

```tsx title="button-snippet.tsx"
import { cva, type VariantProps } from "class-variance-authority";
import { Button, type ButtonProps, buttonVariants } from "@rakui/button";
import { cn } from "@/lib/utils";

const customButtonVariants = cva("font-semibold shadow-md", {
  extend: buttonVariants,
  variants: {
    radius: {
      full: "rounded-full",
      lg: "rounded-lg",
      md: "rounded-md",
      sm: "rounded-sm",
    },
  },
  defaultVariants: {
    radius: "full",
    variant: "primary",
  },
});

type CustomButtonProps = ButtonProps &
  VariantProps<typeof customButtonVariants> & {
    className?: string;
  };

function CustomButton({ className, radius, variant, ...props }: CustomButtonProps) {
  return (
    <Button
      className={cn(customButtonVariants({ className, radius, variant }))}
      {...props}
    />
  );
}
```

### Interactive States

Style buttons using data attributes:

- **Hover**: `:hover` or `[data-hovered]`
- **Pressed**: `[data-pressed]`
- **Focus**: `:focus-visible` or `[data-focus-visible]`
- **Disabled**: `[data-disabled]`
- **Pending**: `[data-pending]`
