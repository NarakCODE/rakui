## Layout

Compose the toolbar from the root, action items, and separators.

```tsx title="selection-toolbar-basic.tsx"
"use client";

import { Bold, Italic } from "lucide-react";
import * as React from "react";
import {
  SelectionToolbar,
  SelectionToolbarItem,
  SelectionToolbarSeparator,
} from "@/components/ui/selection-toolbar";

export function EditorSelectionToolbar() {
  const containerRef = React.useRef<HTMLDivElement>(null);

  return (
    <div
      ref={containerRef}
      contentEditable
      suppressContentEditableWarning
      className="rounded-lg border bg-card p-6"
    >
      <p>Select this text to open the toolbar.</p>

      <SelectionToolbar container={containerRef}>
        <SelectionToolbarItem onSelect={() => {}}>
          <Bold className="size-4" />
        </SelectionToolbarItem>
        <SelectionToolbarItem onSelect={() => {}}>
          <Italic className="size-4" />
        </SelectionToolbarItem>
        <SelectionToolbarSeparator />
      </SelectionToolbar>
    </div>
  );
}
```

## Usage

### Basic Usage

Attach the toolbar to a selectable container and provide actions with `SelectionToolbarItem`.

```tsx title="selection-toolbar-basic-usage.tsx"
<SelectionToolbar container={containerRef}>
  <SelectionToolbarItem onSelect={() => wrapSelection("strong")}>
    <Bold className="size-4" />
  </SelectionToolbarItem>
  <SelectionToolbarItem onSelect={() => wrapSelection("em")}>
    <Italic className="size-4" />
  </SelectionToolbarItem>
</SelectionToolbar>
```

### Track Selected Text

Use `onSelectionChange` when you need analytics, counters, or contextual UI tied to the active selection.

<ComponentTabs name="selection-toolbar-info-demo" />

```tsx title="selection-toolbar-selection-info.tsx"
<SelectionToolbar
  container={containerRef}
  onSelectionChange={(text) => {
    setSelectedText(text);
    setWordCount(text.trim().split(/\s+/).filter(Boolean).length);
  }}
>
  <SelectionToolbarItem onSelect={onCopy}>
    <Copy className="size-4" />
  </SelectionToolbarItem>
</SelectionToolbar>
```

### Custom Positioning

Change `side`, `align`, and collision options when the default top-center placement is not ideal for your editor layout.

```tsx title="selection-toolbar-positioning.tsx"
<SelectionToolbar
  container={containerRef}
  side="bottom"
  align="start"
  sideOffset={12}
  collisionPadding={8}
>
  <SelectionToolbarItem onSelect={onCopy}>
    <Copy className="size-4" />
  </SelectionToolbarItem>
</SelectionToolbar>
```
